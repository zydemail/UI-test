define(function(require, exports, module) {
    // require('../lib/glue-1.1.2.min');
    APP.init({});
 
    var a = 1;
 
    module.exports = {
        b: 2,
        /**
         * 获取两个日期相差天数
         */
        getDateDiff : function(date1, date2) {
            if (!( date1 instanceof Date)) {
                date1 = new Date(+date1);
            }
            if (!( date2 instanceof Date)) {
                date2 = new Date(+date2);
            }
 
            function getYMD(date) {
                return {
                    "y" : date.getFullYear(), //年
                    "m" : date.getMonth(), //月
                    "d" : date.getDate(), //日
                };
            }
 
            var o1 = getYMD(date1);
            var o2 = getYMD(date2);
            var oDate1 = new Date(o1.y, o1.m, o1.d);
            var oDate2 = new Date(o2.y, o2.m, o2.d);
            //把相差的毫秒数转换为天数  1000*60*60*24
            var iDays = (Math.abs(oDate2 - oDate1) / 8.64e7) >> 0;
            return iDays;
        },
    };
 
});
